import React, { useState } from 'react';

interface WordLengthSelectorProps {
  onLengthChange?: (length: number) => void;
}

const WordLengthSelector: React.FC<WordLengthSelectorProps> = ({ onLengthChange }) => {
  const [wordLength, setWordLength] = useState<number>(5);

  const handleChange = (event: React.ChangeEvent<HTMLInputElement>) => {
    const value = parseInt(event.target.value, 10);
    setWordLength(value);
    if (onLengthChange) {
      onLengthChange(value);
    }
  };

  return (
    <div style={{ display: 'flex', justifyContent: "space-between", alignItems: 'center', gap: '10px' }}>
      <label style={{ fontSize: '16px' }}>Välj ordlängd:</label>
      <input
        type="range"
        min="3"
        max="15"
        value={wordLength}
        onChange={handleChange}
        style={{width: '120px', color: '#4CAF50' }}
      />
      <span style={{ width: '30px', fontSize: '16px', textAlign: 'right' }}>{wordLength}</span>
    </div>
  );
};

export default WordLengthSelector;
